from flask import Flask, request, jsonify
import face_recognition

app = Flask(__name__)

# Load known face image (you can add more later)
known_image = face_recognition.load_image_file("known/anjan.jpg")
known_encoding = face_recognition.face_encodings(known_image)[0]

@app.route('/recognize', methods=['POST'])
def recognize_face():
    file = request.files['image']
    unknown_image = face_recognition.load_image_file(file)

    unknown_encodings = face_recognition.face_encodings(unknown_image)
    if not unknown_encodings:
        return jsonify({"success": False, "message": "No face detected"})

    result = face_recognition.compare_faces([known_encoding], unknown_encodings[0])
    match = result[0]

    return jsonify({
        "success": True,
        "matched": match,
        "name": "Anjan" if match else "Unknown"
    })

def home():

if __name__ == '__main__':
    app.run(host='0.0.0.0', port=10000)
